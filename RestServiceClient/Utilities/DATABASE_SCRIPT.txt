CREATE TABLE METHOD (
    ID INT PRIMARY KEY,
    TYPE VARCHAR(15),
    DESCRIPTION VARCHAR(255)
);

CREATE TABLE HEADER(
    ID INT AUTO_INCREMENT PRIMARY KEY,
    HEADER_KEY VARCHAR(255),
    HEADER_VALUE VARCHAR(255)
);

CREATE TABLE REQUEST (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    NAME VARCHAR(255),
    ADDRESS VARCHAR(255),
    BODY VARCHAR(255),
    METHOD_ID INT,
    CONSTRAINT FK_METHOD_ID FOREIGN KEY (METHOD_ID) REFERENCES METHOD(ID),
    HEADER_ID INT,
    CONSTRAINT FK_HEADER_ID FOREIGN KEY (HEADER_ID) REFERENCES HEADER(ID)
);

CREATE TABLE QUERY_STRINGS(
    ID INT AUTO_INCREMENT PRIMARY KEY,
    QUERY_KEY VARCHAR(255),
    QUERY_VALUE VARCHAR(255),
    REQUEST_ID INT,
    CONSTRAINT FK_REQUEST_ID FOREIGN KEY (REQUEST_ID) REFERENCES REQUEST(ID)
);
